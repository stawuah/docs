## Local development

Clone the repo and install dependencies.

```bash
git clone https://github.com/stawuah/swiftui-rn
cd swiftui-rn
npm install
```

Build the TypeScript source:

```bash
npm run build # compile once
npm run dev # watch mode
```

The compiled output lands in `dist/` — this is what gets published to npm.

## Project structure

```text
swiftui-rn/
├── src/
│ ├── types.ts ← base ViewNode class, NodeProps interface, toRNStyle()
│ ├── nodes.ts ← VStackNode, TextNode, ButtonNode, ImageNode, etc.
│ ├── renderer.tsx ← <Render node={...} /> component
│ └── index.ts ← public factory functions + re-exports
├── examples/
│ └── SendScreen.tsx ← full production example
├── dist/ ← compiled output (gitignored)
├── package.json
├── tsconfig.json
└── README.md
```

## How a node becomes a component

<Steps>
  <Step title="You call a factory">
    ```ts
    const btn = Button('Send', onPress).bg('#16a34a').padding(18).rounded(14)
    ```

    This creates a `ButtonNode` instance and sets props on it via chainable methods.
  </Step>
  <Step title="Props are stored as plain data">
    ```ts
    btn._props // { text: 'Send', onPress: fn, backgroundColor: '#16a34a', padding: 18, borderRadius: 14 }
    btn._type // 'Button'
    ```

    No rendering has happened yet. The node is just a data object.
  </Step>
  <Step title="Render walks the tree">
    When `<Render node={btn} />` mounts, it calls `toRNStyle()` on the node's props, then returns the correct React Native component:

    ```tsx
    // What Render outputs for a ButtonNode:
    <TouchableOpacity
    onPress={props.onPress}
    activeOpacity={0.8}
    style={toRNStyle(props)}
    >
    <Text style={{ color: props.color, fontSize: props.fontSize }}>
    {props.text}
    </Text>
    </TouchableOpacity>
    ```
  </Step>
</Steps>

## Adding a new primitive

1. Add a new type to the `NodeType` union in `src/types.ts`:

```ts
export type NodeType =
| 'VStack' | 'HStack' | ...
| 'Badge' // ← your new type
```

2. Create a subclass in `src/nodes.ts`:

```ts
export class BadgeNode extends ViewNode<BadgeNode> {
constructor(label: string) {
super('Badge', { text: label })
}
variant(v: 'success' | 'error' | 'info'): BadgeNode {
const colors = { success: '#16a34a', error: '#ef4444', info: '#3b82f6' }
this._props.backgroundColor = colors[v]
return this
}
}
```

3. Add the factory in `src/index.ts`:

```ts
export const Badge = (label: string) => new BadgeNode(label)
```

4. Handle it in `src/renderer.tsx`:

```tsx
case 'Badge':
return (
<View style={[{ borderRadius: 999, paddingHorizontal: 10, paddingVertical: 4 }, style]}>
<Text style={{ color: '#fff', fontSize: 12, fontWeight: '600' }}>{p.text}</Text>
</View>
)
```

## Publishing to npm

```bash
# 1. Build
npm run build

# 2. Login (first time only)
npm login

# 3. Publish
npm publish --access public
```

<Warning>
  Update the `"author"` and `"repository"` fields in `package.json` before publishing. Also bump `"version"` using semver for every subsequent release.
</Warning>

### Versioning guide

| Change type              | Version bump | Example       |
| ------------------------ | ------------ | ------------- |
| Bug fix                  | Patch        | 0.1.0 → 0.1.1 |
| New modifier / primitive | Minor        | 0.1.0 → 0.2.0 |
| Breaking API change      | Major        | 0.1.0 → 1.0.0 |
